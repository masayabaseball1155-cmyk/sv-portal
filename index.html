// ============================================= // SV加盟店管理ポータル - GAS API v3 // ============================================= const SHEET_GOAL = "加盟店目標"; const SHEET_REAL = "実数値"; const SHEET_TASK = "タスク"; const SHEET_MIKOMI = "見込み数値"; const SHEET_STAFF = "スタッフランク"; const SHEET_SALES = "スタッフ売上（9~2月）"; function doGet(e) { try { return json({ stores: getStores(), tasks: getTasks(), staffRanks: getStaffRanks(), staffSales: getStaffSales() }); } catch(err) { return json({ error: err.toString() }); } } function doPost(e) { try { const data = JSON.parse(e.postData.contents); if (data.action === "updateTask") { updateTask(data.store, data.task); return json({ ok: true }); } if (data.action === "updateStaffRank") { updateStaffRank(data.store, data.staff, data.rank, data.score); return json({ ok: true }); } if (data.action === "deleteStaff") { deleteStaff(data.store, data.staff); return json({ ok: true }); } return json({ error: "unknown action" }); } catch(err) { return json({ error: err.toString() }); } } function getStores() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const rowsG = ss.getSheetByName(SHEET_GOAL).getDataRange().getValues(); const rowsR = ss.getSheetByName(SHEET_REAL).getDataRange().getValues(); const rowsM = ss.getSheetByName(SHEET_MIKOMI).getDataRange().getValues(); // 実数値マップ const actualMap = {}; for (let i = 1; i < rowsR.length; i++) { const nm = String(rowsR[i][2] || "").trim(); if (nm) actualMap[nm] = rowsR[i]; } // 見込み数値マップ const mikomiMap = {}; for (let i = 1; i < rowsM.length; i++) { const nm = String(rowsM[i][2] || "").trim(); if (nm) mikomiMap[nm] = rowsM[i]; } // 日割り計算用 const today = new Date(); const elapsed = today.getDate() - 1; const daysInMonth = new Date(today.getFullYear(), today.getMonth() + 1, 0).getDate(); const stores = []; for (let i = 1; i < rowsG.length; i++) { const g = rowsG[i]; const sv = String(g[1] || "").trim(); const nm = String(g[2] || "").trim(); // 空行スキップのみ（数式チェックは不要：getValues()は計算済み値を返す） if (!nm || !sv) continue; const a = actualMap[nm] || []; const m = mikomiMap[nm] || []; const tgt = pf(g[5]); const act = pf(a[5]); const pct = tgt > 0 ? round1(act / tgt * 100) : null; // 見込み売上（日割り） const mkSales = elapsed > 0 ? Math.round(act / elapsed * daysInMonth) : null; const mkPct = tgt > 0 && mkSales ? round1(mkSales / tgt * 100) : null; // 見込み客数（同一日割りで統一） const mkTotal = elapsed > 0 ? Math.round(pf(a[8]) / elapsed * daysInMonth) : 0; const mkShinki = elapsed > 0 ? Math.round(pf(a[6]) / elapsed * daysInMonth) : 0; const mkRairai = elapsed > 0 ? Math.round(pf(a[7]) / elapsed * daysInMonth) : 0; stores.push({ sv, name: nm, // 目標 売上目標: tgt, 新規目標: pf(g[6]), 再来目標: pf(g[7]), 総客数目標: pf(g[8]), 客単価目標: pf(g[9]), 回数券売上目標: pf(g[14]), 次回予約率目標: pf(g[13]), ロイヤリティ目標: pf(g[32]), SV売上目標: pf(g[34]), // 実績 実績売上: act, 達成率: pct, 新規実績: pf(a[6]), 再来実績: pf(a[7]), 総客数実績: pf(a[8]), 客単価実績: pf(a[9]), 施術単価実績: pf(a[10]), 新規売上実績: pf(a[11]), 再来売上実績: pf(a[13]), 回数券売上実績: pf(a[15]), 物販売上実績: pf(a[20]), 次回予約率実績: pf(a[30]), ロイヤリティ実績: pf(a[31]), SV売上実績: pf(a[33]), // 見込み（日割り統一計算） 見込み新規: mkShinki, 見込み再来: mkRairai, 見込み総客数: mkTotal, 見込み総客数率: pf(m[9]), 見込み客単価: pf(m[10]) || pf(a[9]), 見込み回数券: pf(m[12]) || mk(a[15], elapsed, daysInMonth), 見込み物販: pf(m[14]) || mk(a[20], elapsed, daysInMonth), 見込みロイヤリティ: mk(a[31], elapsed, daysInMonth), 見込みSV売上: mk(a[33], elapsed, daysInMonth), 見込み売上: mkSales, 見込み達成率: mkPct, }); } return stores; } function getTasks() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const rowsT = ss.getSheetByName(SHEET_TASK).getDataRange().getValues(); const rowsG = ss.getSheetByName(SHEET_GOAL).getDataRange().getValues(); const storeMap = {}; for (let i = 1; i < rowsG.length; i++) { const sv = String(rowsG[i][1] || "").trim(); const nm = String(rowsG[i][2] || "").trim(); if (sv && nm) storeMap[i + 1] = { sv, name: nm }; } const tasks = []; for (let i = 1; i < rowsT.length; i++) { const cell = String(rowsT[i][0] || ""); let sv = "", store = ""; const mm = cell.match(/!B(\d+)/); if (mm) { const si = storeMap[parseInt(mm[1])]; if (si) { sv = si.sv; store = si.name; } } else if (cell && !cell.startsWith("=")) { sv = cell; store = String(rowsT[i][1] || ""); } if (!store) continue; tasks.push({ sv, store, 施策: rowsT[i][2] || "", 次回予約特典: rowsT[i][3] || "", ミニモ導入: rowsT[i][4] === true || rowsT[i][4] === "TRUE" || rowsT[i][4] === "ok", スタッフMTG: rowsT[i][5] === true || rowsT[i][5] === "TRUE" || rowsT[i][5] === "ok", オーナーMTG: rowsT[i][6] === true || rowsT[i][6] === "TRUE" || rowsT[i][6] === "ok", HPBミーティング: rowsT[i][7] === true || rowsT[i][7] === "TRUE" || rowsT[i][7] === "ok", 店舗集計分析: rowsT[i][8] === true || rowsT[i][8] === "TRUE" || rowsT[i][8] === "ok", 未実行タスク: rowsT[i][9] || "", }); } return tasks; } function updateTask(storeName, taskData) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const wsT = ss.getSheetByName(SHEET_TASK); const rowsT = wsT.getDataRange().getValues(); const rowsG = ss.getSheetByName(SHEET_GOAL).getDataRange().getValues(); const storeMap = {}; for (let i = 1; i < rowsG.length; i++) { const nm = String(rowsG[i][2] || "").trim(); if (nm) storeMap[nm] = i + 1; } let targetRow = -1; for (let i = 1; i < rowsT.length; i++) { const cell = String(rowsT[i][0] || ""); const mm = cell.match(/!B(\d+)/); if (mm) { if (storeMap[storeName] === parseInt(mm[1])) { targetRow = i + 1; break; } } else if (String(rowsT[i][1] || "") === storeName) { targetRow = i + 1; break; } } if (targetRow < 0) { const sr = storeMap[storeName]; wsT.appendRow([ sr ? `='${SHEET_GOAL}'!B${sr}` : "", sr ? `='${SHEET_GOAL}'!C${sr}` : storeName, taskData.施策 || "", taskData.次回予約特典 || "", taskData.ミニモ導入 ? true : false, taskData.スタッフMTG ? true : false, taskData.オーナーMTG ? true : false, taskData.HPBミーティング ? true : false, taskData.店舗集計分析 ? true : false, taskData.未実行タスク || "", ]); } else { if (taskData.施策 !== undefined) wsT.getRange(targetRow, 3).setValue(taskData.施策 || ""); if (taskData.次回予約特典 !== undefined) wsT.getRange(targetRow, 4).setValue(taskData.次回予約特典 || ""); if (taskData.ミニモ導入 !== undefined) wsT.getRange(targetRow, 5).setValue(taskData.ミニモ導入 ? true : false); if (taskData.スタッフMTG !== undefined) wsT.getRange(targetRow, 6).setValue(taskData.スタッフMTG ? true : false); if (taskData.オーナーMTG !== undefined) wsT.getRange(targetRow, 7).setValue(taskData.オーナーMTG ? true : false); if (taskData.HPBミーティング !== undefined) wsT.getRange(targetRow, 8).setValue(taskData.HPBミーティング ? true : false); if (taskData.店舗集計分析 !== undefined) wsT.getRange(targetRow, 9).setValue(taskData.店舗集計分析 ? true : false); if (taskData.未実行タスク !== undefined) wsT.getRange(targetRow, 10).setValue(taskData.未実行タスク || ""); } } function pf(v) { const n = parseFloat(v); return isNaN(n) ? 0 : n; } function mk(v, elapsed, dim) { const n = pf(v); return (!n || elapsed <= 0) ? 0 : Math.round(n / elapsed * dim); } function round1(v) { return Math.round(v * 10) / 10; } function getStaffRanks() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let ws = ss.getSheetByName(SHEET_STAFF); // シートが無ければ自動作成 if (!ws) { ws = ss.insertSheet(SHEET_STAFF); ws.getRange(1, 1, 1, 4).setValues([["店舗名", "スタッフ名", "ランク", "点数"]]); } const rows = ws.getDataRange().getValues(); const staffs = []; for (let i = 1; i < rows.length; i++) { const store = String(rows[i][0] || "").trim(); const name = String(rows[i][1] || "").trim(); const rank = String(rows[i][2] || "").trim(); const score = pf(rows[i][3]); if (store && name) staffs.push({ store, name, rank, score, row: i + 1 }); } return staffs; } function getStaffSales() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const ws = ss.getSheetByName(SHEET_SALES); if (!ws) return []; const rows = ws.getDataRange().getValues(); const map = {}; for (let i = 1; i < rows.length; i++) { const kubun = String(rows[i][5] || "").trim(); const staffRaw = String(rows[i][13] || "").trim(); const amt = parseFloat(rows[i][12]) || 0; const dateNum = parseFloat(rows[i][1]) || 0; if (kubun !== "施術") continue; if (!staffRaw || staffRaw === "フリー 指名なし") continue; const store = String(rows[i][0] || "").trim(); // シート上で既に短縮名に書き換え済み if (!store) continue; const month = Math.floor(dateNum / 100) % 100; const key = store + "|" + staffRaw; if (!map[key]) map[key] = { store, name: staffRaw, monthly: {} }; map[key].monthly[month] = (map[key].monthly[month] || 0) + amt; } // 月平均売上でランク判定（松:85万以上 / 竹:73〜85万 / 梅:73万以下） const result = []; Object.values(map).forEach(s => { const months = Object.values(s.monthly); if (!months.length) return; const avg = Math.round(months.reduce((a, b) => a + b, 0) / months.length); const rank = avg >= 850000 ? "松" : avg >= 730000 ? "竹" : "梅"; result.push({ store: s.store, name: s.name, avg, rank, monthly: s.monthly, months: months.length }); }); return result.sort((a, b) => b.avg - a.avg); } function updateStaffRank(storeName, staffName, rank, score) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let ws = ss.getSheetByName(SHEET_STAFF); if (!ws) { ws = ss.insertSheet(SHEET_STAFF); ws.getRange(1, 1, 1, 4).setValues([["店舗名", "スタッフ名", "ランク", "点数"]]); } const rows = ws.getDataRange().getValues(); let targetRow = -1; for (let i = 1; i < rows.length; i++) { if (String(rows[i][0] || "").trim() === storeName && String(rows[i][1] || "").trim() === staffName) { targetRow = i + 1; break; } } if (targetRow < 0) { ws.appendRow([storeName, staffName, rank || "", score || ""]); } else { if (rank !== undefined) ws.getRange(targetRow, 3).setValue(rank || ""); if (score !== undefined) ws.getRange(targetRow, 4).setValue(score || ""); } } function deleteStaff(storeName, staffName) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const ws = ss.getSheetByName(SHEET_STAFF); if (!ws) return; const rows = ws.getDataRange().getValues(); for (let i = rows.length - 1; i >= 1; i--) { if (String(rows[i][0] || "").trim() === storeName && String(rows[i][1] || "").trim() === staffName) { ws.deleteRow(i + 1); break; } } } function json(data) { return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON); }
